The _safe_db_value helper is too aggressive — it converts '2' (string) to 2 (int), then asyncpg rejects an int going into a VARCHAR column. We need column-type-aware coercion.

Replace _safe_db_value and the field-setting loop in shred_form with this:


from sqlalchemy import inspect as sa_inspect
from sqlalchemy import Numeric as SANumeric, Integer as SAInteger, String as SAString


def _coerce_for_column(model_instance, field_name, value):
    """Coerce value to match the actual DB column type."""
    if value is None:
        return None
    if isinstance(value, str) and value.strip() == '':
        return None

    try:
        col = sa_inspect(type(model_instance)).columns.get(field_name)
        if col is not None:
            col_type = col.type
            if isinstance(col_type, SAInteger):
                try:
                    return int(value)
                except (ValueError, TypeError):
                    return None
            elif isinstance(col_type, SANumeric):
                try:
                    return float(value)
                except (ValueError, TypeError):
                    return None
            else:
                # VARCHAR / Text / String — always convert to str
                if isinstance(value, (int, float)):
                    return str(value)
    except Exception:
        pass

    return value
And update the field-setting loop inside shred_form:


for field_name, value in item.items():
    if not hasattr(db_item, field_name):
        continue
    coerced = _coerce_for_column(db_item, field_name, value)
    if coerced is None:
        continue  # let DB defaults apply
    setattr(db_item, field_name, coerced)
Also remove the old _safe_db_value function entirely — _coerce_for_column replaces it completely. The _safe_num helper for totals can stay as-is.
